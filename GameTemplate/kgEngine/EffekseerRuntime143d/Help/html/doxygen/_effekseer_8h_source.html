309 inline int32_t ConvertUtf16ToUtf8( int8_t* dst, int32_t dst_size, const int16_t* src )
310 {
311  int32_t cnt = 0;
312  const int16_t* wp = src;
313  int8_t* cp = dst;
314 
315  if (dst_size == 0) return 0;
316 
317  dst_size -= 3;
318 
319  for (cnt = 0; cnt < dst_size; )
320  {
321  int16_t wc = *wp++;
322  if (wc == 0)
323  {
324  break;
325  }
326  if ((wc & ~0x7f) == 0)
327  {
328  *cp++ = wc & 0x7f;
329  cnt += 1;
330  } else if ((wc & ~0x7ff) == 0)
331  {
332  *cp++ = ((wc >> 6) & 0x1f) | 0xc0;
333  *cp++ = ((wc) & 0x3f) | 0x80;
334  cnt += 2;
335  } else {
336  *cp++ = ((wc >> 12) & 0xf) | 0xe0;
337  *cp++ = ((wc >> 6) & 0x3f) | 0x80;
338  *cp++ = ((wc) & 0x3f) | 0x80;
339  cnt += 3;
340  }
341  }
342  *cp = '\0';
343  return cnt;
344 }

353 inline int32_t ConvertUtf8ToUtf16( int16_t* dst, int32_t dst_size, const int8_t* src )
354 {
355  int32_t i, code;
356  int8_t c0, c1, c2;
357 
358  if (dst_size == 0) return 0;
359 
360  dst_size -= 1;
361 
362  for (i = 0; i < dst_size; i++)
363  {
364  int16_t wc;
365 
366  c0 = *src++;
367  if (c0 == '\0')
368  {
369  break;
370  }
371  // UTF8からUTF16に変換
372  code = (uint8_t)c0 >> 4;
373  if (code <= 7)
374  {
375  // 8bit文字
376  wc = c0;
377  }
378  else if (code >= 12 && code <= 13)
379  {
380  // 16bit文字
381  c1 = *src++;
382  wc = ((c0 & 0x1F) << 6) | (c1 & 0x3F);
383  }
384  else if (code == 14)
385  {
386  // 24bit文字
387  c1 = *src++;
388  c2 = *src++;
389  wc = ((c0 & 0x0F) << 12) | ((c1 & 0x3F) << 6) | (c2 & 0x3F);
390  }
391  else
392  {
393  continue;
394  }
395  dst[i] = wc;
396  }
397  dst[i] = 0;
398  return i;
399 }
